# 01_findTheBug

- 문자열을 입력받아 버그('#')의 인덱스를 리턴
  - '#' 기호가 없는 경우 `undefined`를 리턴

```
function findTheBug(word) {
  for(let i =0 ; word.length>i; i++){
    if(word[i]==='#'){
      return i;
    }
  }
  return undefined;
}
```

# 02_sumTo

- 수를 입력받아 0부터 해당 수까지의 합을 리턴

```
function sumTo(num) {
  let result =0;
  for(let i =0; num>=i; i++){
    result = result + i;
  } return result;
}
```

# 03_isOdd

- 수를 입력받아 홀수인지 여부를 리턴

```
function isOdd(num) {
  if(num < 0){
    num = -num;
  }
  while(num >= 0){
    if(num === 0){
      return false;
    } else if(num === 1){
      return true;
    }
     num = num-2;
  }
}
```

# 04_factorial

- 수를 입력받아 n-factorial(n!) 값을 리턴

```
function factorial(num) {
  let result = 1;
  if(num === 0){
    return 1;
  }

  for (let i =1; num >= i; i++){
    result = result * i;
  }
  return result;
}

```

# 05_repeatString

- 문자열과 수를 입력받아 반복된 문자열을 리턴

```
function repeatString(str, num) {
  let result= '';

  if(num === 0){
    result = '';
  }

  for(let i = 0; num > i; i++){
    result = result + str;
  }
  return result;
}
```

# 06_makeDigits

- 수(`num`)를 입력받아 1부터 `num`까지의 정수로 구성된 문자열을 리턴

```
function makeDigits(num) {
  let result = '';

  for(let i = 1; num>= i; i++){
    result = result + String(i);
  }
  return result;
}
```

# 08_makeOddDigits

- 수(`num`)를 입력받아 1을 포함하여 `num`개의 홀수로 구성된 문자열을 리턴

```
function makeOddDigits(num) {
  let result = '';

  let count = 0;
  let i = 1;
  while (count < num) {
    result = result + String(i);
    i = i + 2;
    count++;
  }

  return result;
}
```

# 09_makeMultiplesOfDigit

- 수를 입력받아 1부터 해당 수까지의 수 중에서 3의 배수로만 구성된 문자열을 리턴

```
function makeMultiplesOfDigit(num) {
  let result= '';

  for(let i = 3; i <= num; i+=3 ){
    result = result + i;
  }
  return result;
}
```

# 10_makeMultiplesOfDigit2

- 두 개의 수를 입력받아 두 수를 포함해 두 수 사이의 수 중 2의 배수의 개수를 리턴

```
function makeMultiplesOfDigit2(num1, num2) {
  let small =num1;
  let big = num2;
  let count = 0;

  if(num1 > num2){
    small = num2;
    big = num1;
  }

  if(small === 0){
    small = 1
  }

  for(let i = small; i <= big; i++ ){
    if(i % 2 === 0){
      count += 1
    }
  }
  return count;
}
```

# 11_countCharacter

- 문자열과 문자를 입력받아 문자열에서 문자(letter)가 등장하는 횟수를 리턴

```
function countCharacter(str, letter) {
  let count = 0;

  for(let i =0; i <= str.length; i++){
    if(str[i]===letter){
      count= count+1;
    }
  }
  return count;
}

```
